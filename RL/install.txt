### Install mujoco

## Get MuJoCo
mkdir ~/.mujoco
cd .mujoco
# Download mujoco210 archive and extract in .mujoco

## Set MuJoCo environment variables
gedit ~/.bashrc
export LD_LIBRARY_PATH=/home/${USER}/.mujoco/mujoco210/bin${LD_LIBRARY_PATH:+:${LD_LIBRARY_PATH}}
export MUJOCO_KEY_PATH=/home/${USER}/.mujoco${MUJOCO_KEY_PATH}

## Test MuJoCo
cd ~/.mujoco/mujoco210/bin
./simulate ../model/humanoid.xml


### Install mujoco-py (to cmmunicate with MuJoCo in Python)

## Create virtual environment
conda create -n mujoco-gym
conda activate mujoco-gym

## Install mujoco-py
cd ~/.mujoco
git clone https://github.com/openai/mujoco-py.git
cd mujoco-py 
pip3 install -e .

## If error "no file named 'patchelf'", do:
# Download: https://nixos.org/releases/patchelf/patchelf-0.9/patchelf-0.9.tar.gz
cd patchelf-0.9
./configure
make
sudo make install
patchelf --version
